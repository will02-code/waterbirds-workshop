make_plot <- function(model, is_hu = TRUE, response_var) {
  direction <- describe_posterior(
    model,
    component = "conditional",
    rope_ci = 1,
    as_p = TRUE
  ) |>
    mutate(component = response_var)
  if (is_hu) {
    hurdle_dir <- describe_posterior(
      model,
      component = "distributional",
      rope_ci = 1,
      rope_range = c(-0.18, 0.18),
      as_p = TRUE
    ) %>%
      mutate(component = "Hurdle")

    # FLIP THE MEDIAN for the hurdle component so direction (p/n/u) matches Presence Probability
    hurdle_dir <- hurdle_dir %>%
      mutate(
        Median = if_else(str_detect(Parameter, "^b_hu_"), Median * -1, Median)
      )

    direction <- direction %>% bind_rows(hurdle_dir)
  }
  direction <- direction %>%
    mutate(
      direction = case_when(
        ROPE_Percentage < 0.025 & Median > 0 ~ "p",
        ROPE_Percentage < 0.025 & Median < 0 ~ "n",
        TRUE ~ "u"
      )
    )

  drws <- model %>%
    gather_draws(`b_.*`, regex = TRUE) %>%
    filter(!str_detect(.variable, "Intercept"))
  new_drws <- drws %>%
    left_join(direction, by = c(".variable" = "Parameter")) %>%
    mutate(
      is_hurdle = str_detect(.variable, "hu_"),
      component = if_else(is_hurdle, "Hurdle", response_var) |>
        factor(levels = c(response_var, "Hurdle")),
      .value = if_else(is_hurdle, .value * -1, .value),
      .value = exp(.value),
      variable_clean = str_replace(.variable, "b_hu_", ""),
      variable_clean = str_replace(variable_clean, "b_", "") %>%
        str_replace_all("_", " ") %>%
        str_to_sentence()
    )
  new_drws <- new_drws %>%
    mutate(direction = factor(direction, levels = c("u", "p", "n")))

  intervals <- new_drws %>%
    group_by(variable_clean, component, direction) %>%
    median_qi(.value, .width = c(0.5, 0.8, 0.95)) %>%
    mutate(colour_group = paste(direction, .width, sep = "_")) %>%
    arrange(desc(.width))

  ggplot(intervals, aes(y = variable_clean, colour = colour_group)) +
    geom_linerange(
      aes(xmin = .lower, xmax = .upper),
      linewidth = 3,
      position = position_dodge(width = 0)
    ) +
    scale_colour_manual(
      values = c(
        "u_0.95" = "grey85",
        "u_0.8" = "grey60",
        "u_0.5" = "grey30",
        "p_0.95" = "#a1d99b",
        "p_0.8" = "#41ab5d",
        "p_0.5" = "#006d2c",
        "n_0.95" = "#fc9272",
        "n_0.8" = "#ef3b2c",
        "n_0.5" = "#99000d"
      )
    ) +
    geom_vline(xintercept = 1, linetype = "dashed", alpha = 0.5) +
    facet_wrap(~component, scales = "free_x") +
    scale_y_discrete(drop = TRUE) +
    scale_x_log10(
      breaks = scales::breaks_log()
    ) +
    labs(
      x = "Posterior Estimate (Odds ratio for hurdle, rate ratio for count)",
      y = "Predictor Variable"
    ) +
    my_theme(legend.position = "none")
}


make_table <- function(model, is_hu = TRUE, as_text = FALSE) {
  
  direction <- describe_posterior(
    model,
    component = "conditional",
    rope_ci = 1
  ) |>
    mutate(component = "Count")

  if (is_hu) {
    direction <- direction |>
      bind_rows(
        describe_posterior(
          model,
          component = "distributional",
          rope_ci = 1,
          rope_range = c(-0.18, 0.18)
        ) |>
          mutate(component = "Hurdle") # Updated Label
      )
  }
  direction_clean <- direction %>%
    mutate(
      is_hurdle = str_detect(Parameter, "^b_hu_"),
      Median = if_else(is_hurdle, Median * -1, Median),
      orig_low = CI_low,
      CI_low = if_else(is_hurdle, CI_high * -1, CI_low),
      CI_high = if_else(is_hurdle, orig_low * -1, CI_high),
      pd_pct = pd * 100,
      ROPE_Percentage = ROPE_Percentage * 100,
      Median = exp(Median),
      CI_low = exp(CI_low),
      CI_high = exp(CI_high),
      `95% Credible Interval` = sprintf("[%.2f, %.2f]", CI_low, CI_high),
      In_Text_Report = sprintf(
        "(pd = %.1f%%, CI = [%.2f, %.2f], inside ROPE = %.1f%%)",
        pd_pct,
        CI_low,
        CI_high,
        ROPE_Percentage
      ),
      clean_raw = str_remove_all(Parameter, "^b_hu_|^b_"),

      variable_raw = if_else(
        str_detect(clean_raw, ":"),
        str_extract(clean_raw, ":.*$") |> str_remove(":"),
        clean_raw
      ),

      Clean_Parameter = variable_raw %>%
        str_remove_all("_scaled") %>%
        str_replace_all("_", " ") %>%
        str_replace_all("precip", "Precipitation") %>%
        str_replace_all("swc australia", "Surface water (Australia)") %>%
        str_to_title() %>%
        str_replace_all("3m", "(3-Month)") %>%
        str_replace_all("6m", "(6-Month)") %>%
        str_replace_all("Annual", "Annual") %>%
        str_replace_all("Lagged", "Lagged")
    ) %>%
    filter(
      !(variable_raw |> str_detect("Intercept"))
    ) |>

    # Clean up component naming conventions from bayestestR
    arrange(component, Clean_Parameter)
  # 3. Clean, merge, and parse names

  if (as_text) {
    final_df <- direction_clean %>%
      select(
        component,
        Clean_Parameter,
        Median,
        `95% Credible Interval`,
        pd_pct,
        Rhat,
        ROPE_Percentage,
        In_Text_Report
      )
  } else {
    final_df <- direction_clean %>%
      select(
        component,
        Clean_Parameter,
        Median,
        `95% Credible Interval`,
        pd_pct,
        Rhat,
        ROPE_Percentage
      )
  }

  # 4. Generate the nested Flextable layout
  ft <- final_df %>%
    flextable() %>%
    theme_zebra() %>%

    # Structural Formatting: Vertically merge matching consecutive rows
    # merge_v(j = c("component", "Functional_Group")) %>%

    # Clean up borders on the merged elements for visual clarity
    # fix_border_issues() %>%
    autofit() %>%

    # Standard formatting adjustments
    align(
      j = c(
        "Median",
        "95% Credible Interval",
        "pd_pct",
        "Rhat",
        "ROPE_Percentage"
      ),
      align = "center",
      part = "body"
    ) %>%
    colformat_double(j = "Median", digits = 2) %>%
    colformat_double(j = "pd_pct", digits = 1) %>%
    colformat_double(j = "ROPE_Percentage", digits = 1) %>%
    colformat_double(j = "Rhat", digits = 3) %>%

    set_header_labels(
      component = "Model component",
      Clean_Parameter = "Environmental Predictor",
      Median = "Posterior Median",
      pd_pct = "pd (%)",
      Rhat = "Rhat",
      ROPE_Percentage = "ROPE (%)"
    ) %>%

    # Bold predictors with strong overlapping signals (outside ROPE)
    bold(i = ~ ROPE_Percentage < 2.5)

  # Return both visual table and programmatic raw data frames
  return(list(table = ft, data = direction_clean))
}

Run models

First, we need to set formula for the model, set the priors, and perform a prior predictive check. The models for this analysis are (almost) all hurdle models. This means that in effect, we run two models here. First, a binary model to predict presence/absence of birds, and second, a model to predict abundance when birds are present. I recommend this blog for more details on hurdle models in brms, or Chapter 11 of Statistical Rethinking, which covers zero-inflated models (somewhat similar) in more technical detail.

Set formula

Code 
full_formula <- brmsformula(
  value ~ asset_precip_3m_scaled + asset_precip_annual_scaled + asset_precip_annual_scaled_lagged + 
    valley_precip_3m_scaled +valley_precip_annual_scaled + valley_precip_annual_scaled_lagged + basin_precip_3m_scaled + basin_precip_annual_scaled + basin_precip_annual_scaled_lagged + 
    wetland_flow_3m_scaled + wetland_flow_annual_scaled + wetland_flow_annual_scaled_lagged + 
    total_basin_flow_3m_scaled + total_basin_flow_annual_scaled + total_basin_flow_annual_scaled_lagged + 
    swc_australia_scaled + swc_australia_scaled_lagged +
    (1 | Wetland), # random intercept for wetland
  # Same formula for the hurdle component of the model. 
  hu ~ asset_precip_3m_scaled + asset_precip_annual_scaled + asset_precip_annual_scaled_lagged + 
    valley_precip_3m_scaled +valley_precip_annual_scaled + valley_precip_annual_scaled_lagged + 
    basin_precip_3m_scaled + basin_precip_annual_scaled + basin_precip_annual_scaled_lagged + 
    wetland_flow_3m_scaled +wetland_flow_annual_scaled + wetland_flow_annual_scaled_lagged + 
    total_basin_flow_3m_scaled + total_basin_flow_annual_scaled + total_basin_flow_annual_scaled_lagged + swc_australia_scaled + swc_australia_scaled_lagged +
    (1 | Wetland)
)

Set priors

I use weakly informative priors here, so that sampling is constrained to reasonable parameters (e.g. we know there are rarely 5 million birds at a wetland, and that birds cannot be negative), but not overly restrictive. See here and here

priors <- c(
  set_prior(paste0("normal(", 9, ",", 1, ")"), class = "Intercept"),
  set_prior(paste0("normal(", 0, ",", 0.1, ")"), class = "b"),
  set_prior("exponential(2)", class = "shape"),
  set_prior("normal(1, 1)", class = "Intercept", dpar = "hu"),
  set_prior("normal(0, 0.1)", class = "b", dpar = "hu"),
  set_prior("normal(0, 0.5)", class = "sd", lb = 0),
  set_prior("normal(0, 0.5)", class = "sd", lb = 0, dpar = "hu")
)


m1.1_prior <- brm(
  formula = full_formula,
  family = hurdle_gamma(),
  data = full_data,
  backend = "cmdstanr",
  cores = 4,
  prior = priors,
  sample_prior = "only"
)
Prior predictive check

Here, we want to see that data simulated with just the priors is reasonable, and broadly similar to the observed data. In this case, it’s not really that reasonable, but we can see that the simulated datasets (orange) cover the range of the observed data (blue). I prefer priors that are wide rather than narrow, because they allow the data to have a bit more influence.

pp_check(m1.1_prior, "boxplot") + my_theme()+theme(axis.text.x = element_text(angle = 45, hjust = 1))

Run model

Now we run the model with the data

m1.1 <- brm(
  full_formula,
  family = hurdle_gamma(),
  data = full_data,
  backend = "cmdstanr",
  cores = 4,
  prior = priors,
  chains = 4,
  iter = 4000,
  file = "data/m1.1.rds",
  file_refit = "on_change"
)
Model diagnostics

We check the Rhat values (below 1.1), and ESS (above 1000). We can plot the MCMC chains to check for good mixing. (I don’t run this here, because it takes a long time)
