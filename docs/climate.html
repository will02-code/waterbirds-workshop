Helper functions

Code 
get_climate_predictions <- function(
  wetland,
  model,
  spp_code = NULL,
  fx_group = NULL,
  flow_metadata = combined_flow_metadata,
  climate_projections = valley_climate_projections_with_wetlands,
  basin_flow_stats = basin_flow_mean_sd
) {
  # --- Pre-processing -------------------------------------------------------
  scaling <- combined_flow_metadata |>
    filter(wetlands == wetland) |>
    slice(1) |>
    select(
      wetland_mean_flow_3m,
      wetland_sd_flow_3m,
      wetland_mean_flow_annual,
      wetland_sd_flow_annual
    )

  # --- Build scenario newdata -----------------------------------------------
  nuisance_vars <- c(
    "asset_precip_3m_scaled",
    "asset_precip_annual_scaled",
    "asset_precip_annual_scaled_lagged",
    "valley_precip_3m_scaled",
    "valley_precip_annual_scaled",
    "valley_precip_annual_scaled_lagged",
    "basin_precip_3m_scaled",
    "basin_precip_annual_scaled",
    "basin_precip_annual_scaled_lagged",
    "swc_australia_scaled",
    "swc_australia_scaled_lagged"
  )

  scenarios <- valley_climate_projections_with_wetlands |>
    filter(Wetland == wetland) |>
    arrange(Scenario) |>
    select(Scenario, pct_change) |>
    left_join(
      basin_flow_mean_sd |>
        select(Scenario, basin_flow_zscore_3m, basin_flow_zscore_annual),
      by = "Scenario"
    ) |>
    mutate(
      wetland_flow_3m_scaled = (scaling$wetland_mean_flow_3m * pct_change) /
        scaling$wetland_sd_flow_3m,
      wetland_flow_annual_scaled = (scaling$wetland_mean_flow_annual *
        pct_change) /
        scaling$wetland_sd_flow_annual,
      wetland_flow_annual_scaled_lagged = wetland_flow_annual_scaled,
      total_basin_flow_3m_scaled = basin_flow_zscore_3m,
      total_basin_flow_annual_scaled = basin_flow_zscore_annual,
      total_basin_flow_annual_scaled_lagged = basin_flow_zscore_annual,
      Wetland = wetland
    ) |>
    select(-basin_flow_zscore_3m, -basin_flow_zscore_annual)

  pred_newdata <- bind_rows(
    tibble(
      Scenario = "Baseline",
      pct_change = NA_real_,
      wetland_flow_3m_scaled = 0,
      wetland_flow_annual_scaled = 0,
      wetland_flow_annual_scaled_lagged = 0,
      total_basin_flow_3m_scaled = 0,
      total_basin_flow_annual_scaled = 0,
      total_basin_flow_annual_scaled_lagged = 0,
      Wetland = wetland
    ),
    scenarios
  )

  pred_newdata[nuisance_vars] <- 0

  if (!is.null(spp_code)) {
    pred_newdata$spp_code <- spp_code
  }
  if (!is.null(fx_group)) {
    pred_newdata$fx_group <- fx_group
  }

  # --- Posterior draws ------------------------------------------------------
  draws <- posterior_epred(model, newdata = pred_newdata, re_formula = NULL) |>
    t() |>
    as_tibble() |>
    mutate(scenario = pred_newdata$Scenario) |>
    pivot_longer(-scenario, names_to = ".draw", values_to = ".value")

  # --- Percent change (shared computation) ----------------------------------
  draws_with_pct <- draws |>
    group_by(.draw) |>
    mutate(
      baseline_val = .value[scenario == "Baseline"],
      percent_change = (.value - baseline_val) / baseline_val * 100
    ) |>
    ungroup()

  

  summary_results <- draws_with_pct |>
    group_by(scenario) |>
    median_hdci(percent_change, .value) |>
    mutate(wetland_name = wetland)
  if (!is.null(spp_code)) {
    summary_results$spp_code <- spp_code
  }
  if (!is.null(fx_group)) {
    summary_results$fx_group <- fx_group
  }
  list(
    summary = summary_results
  )
}

    na.omit() # Scalars for each wetland, based on the valley it is located in for each climate scenario

basin_flow_mean_sd <- read.csv("data/basin_flow_mean_sd.csv") # This is the mean and sd for the basin flow, which we use for scaling the basin flow variables in the climate data
data(assets)
assets_coords <- assets %>%
  # Ensure we have the center point of each polygon
  mutate(centroid_lat = st_coordinates(st_centroid(geom))[, 2]) %>%
  st_drop_geometry() %>%
  # Sort by latitude descending (North to South)
  arrange(desc(centroid_lat)) %>%
  pull(Wetland) %>%
  str_to_lower()

All wetlands

We wrap this all in a function - get_climate_predictions() and loop through the wetlands

wetlands <- valley_climate_projections_with_wetlands |> 
  pull(Wetland) %>%
  unique() 

overall_abund <- readRDS("data/m1.1.rds")
results <- vector("list", length(wetlands))

for (i in 1:length(wetlands)) {
  # print(wetlands[i])
  
  results[[i]] <- get_climate_predictions(
    wetland = wetlands[i],
    model = overall_abund
  )
}

combined_df_overall <- purrr::map(results, "summary") %>% 
  bind_rows()
